#include <iostream>
using namespace std;

void InsertSort(int a[],int n)
{
	int temp,j;

	for(int i=0;i<n-1;i++)
	{
		temp=a[i+1];
		j=i;
		while(j>=0 && temp<a[j])
		{
			a[j+1]=a[j];
			j--;
		}
		a[j+1]=temp;
	}
}

void printArr(int a[],int n)
{
	for(int i=0;i<n;i++)
	{
		cout<<a[i]<<" ";
	}
	cout<<endl;
}

int main()
{
	int a[10]={5,4,3,2,1,0,9,8,4,-3};
	InsertSort(a,10);
	printArr(a,10);
}
